export class StoryboardError extends Error {
    constructor(
        message: string,
        public code: string,
        public details?: any,
        public suggestions?: string[]
    ) {
        super(message);
        this.name = 'StoryboardError';
    }
}

// 使用
try {
    const panelResults = await extractStoryboard(...);
} catch (error) {
    if (error instanceof StoryboardError) {
        toast.error(error.message, {
            description: error.suggestions?.join('\n') || '请稍后重试',
            action: error.code === 'TIMEOUT' ? {

                onClick: () => window.open('/docs/timeout-fix')
            } : undefined
        });
    } else {

            description: '请联系技术支持'
        });

export function ListView({ panels }: ListViewProps) {
