# 分镜功能改进建议 - 实施方案

## 📋 执行摘要

基于全面代码审查，发现分镜功能整体架构优秀（⭐⭐⭐⭐），但存在以下核心问题需要解决：

### 🔴 高优先级问题（1-2天）

4. **代码重复** - 提示词生成、映射逻辑重复
5. **错误处理不完善** - 错误信息不够详细
6. **性能优化** - 大数据量时需要优化

### 🟢 低优先级问题（1-2周）
7. **文档不足** - 缺少使用示例和架构文档
8. **测试覆盖率低** - 缺少单元测试和集成测试

**步骤 3: 创建数据迁移工具**

```typescript
// src/app/utils/migrations/storyboardMigrations.ts

import { getShotTypeByCN, getCameraAngleByCN, getCameraMovementByCN } from '../../constants/cinematography';
import type { StoryboardPanel } from '../../types';

export interface MigrationResult {
    success: boolean;
    migratedCount: number;
    errors: string[];
}

/**
 * 迁移单个分镜面板
 */
export function migratePanelV1toV2(oldPanel: any): StoryboardPanel {
    const migrated: any = { ...oldPanel };
    
    // 1. 统一景别字段
    if (oldPanel.shotSize) {
        migrated.shot = oldPanel.shotSize;
        delete migrated.shotSize;
    } else if (typeof oldPanel.shot === 'string') {
        const code = getShotTypeByCN(oldPanel.shot);
        migrated.shot = code || 'MS'; // 默认中景
    }
    
    // 2. 统一角度字段
    if (oldPanel.cameraAngle) {
        migrated.angle = oldPanel.cameraAngle;
        delete migrated.cameraAngle;
    } else if (typeof oldPanel.angle === 'string') {
        const code = getCameraAngleByCN(oldPanel.angle);
        migrated.angle = code || 'EYE_LEVEL';
    }
    
    // 3. 统一运动字段
    if (oldPanel.movementType) {
        migrated.movement = oldPanel.movementType;
        delete migrated.movementType;
    } else if (oldPanel.cameraMovement) {
        const code = getCameraMovementByCN(oldPanel.cameraMovement);
        migrated.movement = code || 'STATIC';
        delete migrated.cameraMovement;
    }
    
    // 4. 统一提示词字段
    if (oldPanel.aiPrompt) {
        migrated.imagePrompt = oldPanel.aiPrompt;
        delete migrated.aiPrompt;
    }
    if (oldPanel.aiVideoPrompt) {
        migrated.videoPrompt = oldPanel.aiVideoPrompt;
        delete migrated.aiVideoPrompt;
    }
    if (oldPanel.generatedImage) {
        migrated.generatedImageUrl = oldPanel.generatedImage;
        delete migrated.generatedImage;
    }
    
    return migrated as StoryboardPanel;
}

/**
 * 批量迁移分镜
 */
export function migrateStoryboard(oldStoryboard: any): { storyboard: any; result: MigrationResult } {
    const errors: string[] = [];
    let migratedCount = 0;
    
    const migratedPanels = oldStoryboard.panels.map((panel: any, index: number) => {
        try {
            const migrated = migratePanelV1toV2(panel);
            migratedCount++;
            return migrated;
        } catch (error) {
            errors.push(`面板 ${index + 1} (${panel.id}): ${error}`);
            return panel; // 保留原数据
        }
    });
    
    return {
        storyboard: {
            ...oldStoryboard,
            panels: migratedPanels,
            version: 2, // 标记版本
            migratedAt: new Date().toISOString(),
        },
        result: {
            success: errors.length === 0,
            migratedCount,
            errors,
        }
    };
}
```

// src/app/hooks/queries/useStoryboard.ts

import { migrateStoryboard } from '../../utils/migrations/storyboardMigrations';

export function useStoryboard(chapterId: string | undefined) {
    return useQuery({
        queryKey: queryKeys.storyboard.byChapter(chapterId!),
        queryFn: async () => {
            const data = await storyboardStorage.getByChapterId(chapterId!);
            
            // 🆕 自动迁移旧数据
            if (data && (!data.version || data.version < 2)) {
                console.log('[useStoryboard] 检测到旧版本数据，开始迁移...');
                const { storyboard, result } = migrateStoryboard(data);
                
                if (result.success) {
                    console.log(`[useStoryboard] 迁移成功：${result.migratedCount} 个分镜`);
                    // 自动保存迁移后的数据
                    await storyboardStorage.save(storyboard);
                    return storyboard;
                } else {
                    console.warn('[useStoryboard] 迁移部分失败：', result.errors);
                    return storyboard; // 仍然返回迁移后的数据
                }
            }
            
            return data;
        },
        enabled: !!chapterId,
    });
}

### 2. 添加字段验证

```typescript
// src/app/utils/validation/storyboardValidation.ts

import { z } from 'zod';
import { SHOT_TYPES, CAMERA_ANGLES, CAMERA_MOVEMENTS } from '../../constants/cinematography';

// Zod Schema
export const StoryboardPanelSchema = z.object({
    id: z.string().min(1, '分镜 ID 不能为空'),
    panelNumber: z.number().int().positive('分镜编号必须为正整数'),
    sceneId: z.string().min(1, '场景 ID 不能为空'),
    description: z.string().min(10, '画面描述至少 10 字'),
    dialogue: z.string().optional(),
    
    shot: z.enum(Object.keys(SHOT_TYPES) as [string, ...string[]], {
        errorMap: () => ({ message: '无效的景别类型' })
    }),
    angle: z.enum(Object.keys(CAMERA_ANGLES) as [string, ...string[]], {
        errorMap: () => ({ message: '无效的角度类型' })
    }),
    movement: z.enum(Object.keys(CAMERA_MOVEMENTS) as [string, ...string[]], {
        errorMap: () => ({ message: '无效的运动类型' })
    }),
    
    duration: z.number().min(0.5).max(60, '时长必须在 0.5-60 秒之间').optional(),
    characters: z.array(z.string()),
    props: z.array(z.string()),
    notes: z.string(),
    
    imagePrompt: z.string().optional(),
    videoPrompt: z.string().optional(),
    generatedImageUrl: z.string().url('无效的图片 URL').optional(),
    isImageGenerating: z.boolean().optional(),
    
    // ... 其他字段
});

export type ValidatedStoryboardPanel = z.infer<typeof StoryboardPanelSchema>;

/**
 * 验证单个分镜
 */
export function validatePanel(panel: unknown): { success: boolean; data?: ValidatedStoryboardPanel; errors?: string[] } {
    const result = StoryboardPanelSchema.safeParse(panel);
    
    if (result.success) {
        return { success: true, data: result.data };
    } else {
        const errors = result.error.errors.map(e => `${e.path.join('.')}: ${e.message}`);
        return { success: false, errors };
    }
}

/**
 * 验证分镜数组
 */
export function validatePanels(panels: unknown[]): {
    validPanels: ValidatedStoryboardPanel[];
    invalidPanels: Array<{ index: number; errors: string[] }>;
} {
    const validPanels: ValidatedStoryboardPanel[] = [];
    const invalidPanels: Array<{ index: number; errors: string[] }> = [];
    
    panels.forEach((panel, index) => {
        const result = validatePanel(panel);
        if (result.success && result.data) {
            validPanels.push(result.data);
        } else if (result.errors) {
            invalidPanels.push({ index, errors: result.errors });
        }
    });
    
    return { validPanels, invalidPanels };
}
```

// src/app/pages/StoryboardEditor/hooks/useStoryboardData.ts

import { validatePanels } from '../../../utils/validation/storyboardValidation';

const handleSave = useCallback(async (updatedStoryboard: Storyboard) => {
    try {
        // 🆕 保存前验证
        const { validPanels, invalidPanels } = validatePanels(updatedStoryboard.panels);
        
        if (invalidPanels.length > 0) {
            console.warn('[handleSave] 发现无效分镜：', invalidPanels);
            toast.warning(`发现 ${invalidPanels.length} 个分镜数据异常，已自动修复`);
            // 只保存有效的分镜
            updatedStoryboard.panels = validPanels;
        }
        
        await saveStoryboardMutation.mutateAsync(updatedStoryboard);
        return true;
    } catch (error) {
        console.error('Failed to save storyboard:', error);
        return false;
    }
}, [saveStoryboardMutation]);
