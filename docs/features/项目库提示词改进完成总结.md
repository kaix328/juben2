## 📚 使用指南

### 启用提示词优化

```typescript
const engine = new PromptEngine(directorStyle, {
  enableValidation: true,      // 启用验证
  enableOptimization: true,     // 启用优化
  maxTokens: 150,               // 最大长度
  targetPlatform: 'doubao',     // 目标平台
  outputLanguage: 'mixed',      // 输出语言
});
```

### 使用提示词预览

```typescript
// 在 useImageGeneration hook 中
const {
  enablePromptPreview,
  setEnablePromptPreview,
  promptPreview,
  confirmPreviewAndGenerate,
  cancelPreview,
} = useImageGeneration({ ... });

// 启用预览
setEnablePromptPreview(true);

// 在组件中显示预览对话框
<PromptPreviewDialog
  open={!!promptPreview}
  originalPrompt={promptPreview?.originalPrompt}
  optimizedPrompt={promptPreview?.optimizedPrompt}
  negativePrompt={promptPreview?.negativePrompt}
  platform="doubao"
  onConfirm={confirmPreviewAndGenerate}
  onCancel={cancelPreview}
/>
```

### 使用批量生成

```typescript
// 在 useImageGeneration hook 中
const {
  isBatchGenerating,
  batchTasks,
  batchPaused,
  setBatchPaused,
  setBatchCancelled,
  handleBatchGenerateAll,
} = useImageGeneration({ ... });

// 开始批量生成
handleBatchGenerateAll();

// 显示进度对话框
<BatchGenerationDialog
  open={isBatchGenerating}
  tasks={batchTasks}
  currentIndex={batchProgress.current}
  isPaused={batchPaused}
  onPause={() => setBatchPaused(true)}
  onResume={() => setBatchPaused(false)}
  onCancel={() => setBatchCancelled(true)}
  onClose={() => setIsBatchGenerating(false)}
/>
```

### 平台适配

```typescript
// 获取平台推荐配置
const config = PromptEngine.getPlatformConfig('stable-diffusion');

// 创建引擎
const engine = new PromptEngine(directorStyle, config);

// 格式化提示词
const formatted = PromptEngine.formatForPlatform(prompt, 'midjourney');
```
