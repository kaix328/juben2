```typescript
// src/app/utils/promptEngine.ts

export class PromptEngine {
    // ... 现有代码
    
    /**
     * 添加角色基础信息（名称 + 触发词）
     */
    private addCharacterBasicInfo(character: Character, parts: PromptPart[]): void {
        if (character.name) {
            parts.push({ 
                type: 'subject', 
                value: character.name, 
                weight: 1.2, 
                language: 'zh' 
            });
        }
        
        if (character.triggerWord) {
            parts.push({ 
                type: 'subject', 
                value: character.triggerWord, 
                weight: 1.3, 
                language: 'en' 
            });
        }
    }
    
    /**
     * 添加技术参数（焦距、光圈、景深）
     */
    private addTechnicalParams(panel: StoryboardPanel, parts: PromptPart[]): void {
        if (!panel.lens && !panel.fStop && !panel.depthOfField && !panel.focusPoint) {
            return;
        }
        
        const techParts: string[] = [];
        
        if (panel.lens) {
            techParts.push(`${panel.lens} lens`);
        }
        if (panel.fStop) {
            techParts.push(`shot at ${panel.fStop}`);
        }
        if (panel.depthOfField) {
            const dofMap: Record<string, string> = {
                SHALLOW: 'shallow depth of field, blurred background',
                DEEP: 'deep focus, everything in focus',
                SELECTIVE: 'selective focus, subject isolated',
                NORMAL: 'natural depth of field',
            };
            const dofDesc = dofMap[panel.depthOfField];
            if (dofDesc) {
                techParts.push(dofDesc);
            }
        }
        if (panel.focusPoint) {
            techParts.push(`focus on ${panel.focusPoint}`);
        }
        
        if (techParts.length > 0) {
            parts.push({
                type: 'technical',
                value: techParts.join(', '),
                language: 'en',
            });
        }
    }
    
    /**
     * 添加灯光信息
     */
    private addLightingInfo(lighting: LightingDesign | undefined, parts: PromptPart[]): void {
        if (!lighting) return;
        
        if (lighting.mood) {
            parts.push({
                type: 'lighting',
                value: lighting.mood,
                language: 'zh',
            });
        }
        
        const lightingDetails: string[] = [];
        if (lighting.keyLight) {
            lightingDetails.push(`key light: ${lighting.keyLight}`);
        }
        if (lighting.fillLight) {
            lightingDetails.push(`fill light: ${lighting.fillLight}`);
        }
        if (lighting.backLight) {
            lightingDetails.push(`back light: ${lighting.backLight}`);
        }
        if (lighting.practicalLights && lighting.practicalLights.length > 0) {
            lightingDetails.push(`practical lights: ${lighting.practicalLights.join(', ')}`);
        }
        
        if (lightingDetails.length > 0) {
            parts.push({
                type: 'lighting',
                value: lightingDetails.join(', '),
                language: 'en',
            });
        }
    }
    
    /**
     * 添加景别和角度信息
     */
    private addShotAndAngle(panel: StoryboardPanel, parts: PromptPart[]): void {
        const shotInfo = getShotTypeByCode(panel.shot);
        if (shotInfo) {
            parts.push({ type: 'technical', value: shotInfo.cn, language: 'zh' });
            parts.push({ type: 'technical', value: shotInfo.en, language: 'en' });
        }
        
        const angleInfo = getCameraAngleByCode(panel.angle);
        if (angleInfo) {
            parts.push({ type: 'technical', value: angleInfo.cn, language: 'zh' });
            parts.push({ type: 'technical', value: angleInfo.en, language: 'en' });
        }
    }
    
    // 重构后的方法
    forCharacterFullBody(character: Character, existingPrompt?: string): AdvancedPrompt {
        const parts: PromptPart[] = [];
        
        // 使用提取的方法
        this.addCharacterBasicInfo(character, parts);
        
        if (character.appearance) {
            parts.push({ type: 'appearance', value: character.appearance, weight: 1.1, language: 'zh' });
        }
        
        // 视图类型
        parts.push({ type: 'technical', value: '全身正视图', language: 'zh' });
        parts.push({ type: 'technical', value: 'full body shot, standing pose, front view', language: 'en' });
        parts.push({ type: 'technical', value: '白色背景', language: 'zh' });
        parts.push({ type: 'technical', value: 'white background, simple background', language: 'en' });
        
        if (existingPrompt) {
            parts.push({ type: 'style', value: existingPrompt, language: 'mixed' });
        }
        
        if (this.style) {
            parts.push(...this.styleToPromptParts(this.style));
        }
        
        parts.push(...this.getQualityTags());
        parts.push({ type: 'quality', value: 'character design, reference sheet', language: 'en' });
        
        return this.buildAdvancedPrompt(parts, 'character');
    }
    
    forStoryboardImage(
        panel: StoryboardPanel,
        characters: Character[],
        scenes: Scene[]
    ): AdvancedPrompt {
        const parts: PromptPart[] = [];
        
        // 使用提取的方法
        this.addShotAndAngle(panel, parts);
        this.addTechnicalParams(panel, parts);
        
        if (panel.description) {
            parts.push({ type: 'appearance', value: panel.description, weight: 1.2, language: 'zh' });
        }
        
        if (panel.composition) {
            parts.push({ type: 'style', value: panel.composition, language: 'zh' });
        }
        
        // 角色信息
        if (panel.characters && panel.characters.length > 0) {
            panel.characters.forEach((charName) => {
                const char = characters.find((c) => c.name === charName);
                if (char) {
                    this.addCharacterBasicInfo(char, parts);
                    if (char.appearance) {
                        parts.push({ 
                            type: 'appearance', 
                            value: char.appearance, 
                            weight: 1.1, 
                            language: 'zh' 
                        });
                    }
                }
            });
        }
        
        // 灯光
        this.addLightingInfo(panel.lighting, parts);
        
        if (this.style) {
            parts.push(...this.styleToPromptParts(this.style, true));
        }
        
        parts.push(...this.getQualityTags());
        parts.push({ type: 'quality', value: 'cinematic composition, professional storyboard', language: 'en' });
        
        return this.buildAdvancedPrompt(parts, 'storyboard');
    }
}
```

## 5. 完善错误处理

### 创建自定义错误类

```typescript
// src/app/utils/errors/StoryboardError.ts

export enum ErrorCode {
    // AI 相关
    AI_TIMEOUT = 'AI_TIMEOUT',
    AI_RATE_LIMIT = 'AI_RATE_LIMIT',
    AI_INVALID_RESPONSE = 'AI_INVALID_RESPONSE',
    AI_NETWORK_ERROR = 'AI_NETWORK_ERROR',
    
    // 数据相关
    DATA_VALIDATION_FAILED = 'DATA_VALIDATION_FAILED',
    DATA_NOT_FOUND = 'DATA_NOT_FOUND',
    DATA_SAVE_FAILED = 'DATA_SAVE_FAILED',
    
    // 业务逻辑
    INVALID_PANEL_COUNT = 'INVALID_PANEL_COUNT',
    MISSING_REQUIRED_FIELD = 'MISSING_REQUIRED_FIELD',
    SCENE_NOT_FOUND = 'SCENE_NOT_FOUND',
    
    // 系统错误
    UNKNOWN_ERROR = 'UNKNOWN_ERROR',
}

export interface ErrorSuggestion {
    title: string;
    description: string;
    action?: {
        label: string;
        onClick: () => void;
    };
}

export class StoryboardError extends Error {
    constructor(
        message: string,
        public code: ErrorCode,
        public details?: any,
        public suggestions?: ErrorSuggestion[]
    ) {
        super(message);
        this.name = 'StoryboardError';
        
        // 维护正确的堆栈跟踪
        if (Error.captureStackTrace) {
            Error.captureStackTrace(this, StoryboardError);
        }
    }
    
    /**
     * 创建 AI 超时错误
     */
    static timeout(details?: any): StoryboardError {
        return new StoryboardError(
            'AI 请求超时',
            ErrorCode.AI_TIMEOUT,
            details,
            [
                {
                    title: '减少场景数量',
                    description: '尝试一次处理更少的场景（5-10个）',
                },
                {
                    title: '检查网络连接',
                    description: '确保网络连接稳定',
                },
                {
                    title: '稍后重试',
                    description: 'AI 服务可能繁忙，请等待几分钟后重试',
                },
            ]
        );
    }
    
    /**
     * 创建速率限制错误
     */
    static rateLimit(retryAfter?: number): StoryboardError {
        return new StoryboardError(
            'API 请求频率超限',
            ErrorCode.AI_RATE_LIMIT,
            { retryAfter },
            [
                {
                    title: '等待后重试',
                    description: retryAfter 
                        ? `请等待 ${retryAfter} 秒后重试`
                        : '请稍后重试',
                },
                {
                    title: '减少并发请求',
                    description: '避免同时生成多个分镜',
                },
            ]
        );
    }
    
    /**
     * 创建数据验证错误
     */
    static validation(errors: string[]): StoryboardError {
        return new StoryboardError(
            '数据验证失败',
            ErrorCode.DATA_VALIDATION_FAILED,
            { errors },
            [
                {
                    title: '检查必填字段',
                    description: '确保所有必填字段都已填写',
                },
                {
                    title: '查看详细错误',
                    description: errors.join('\n'),
                },
            ]
        );
    }
}
```

```typescript
            toast.error(error.message, {
                description: error.suggestions?.map(s => `• ${s.title}`).join('\n'),
                duration: 8000,
                action: error.suggestions?.[0]?.action ? {
                    label: error.suggestions[0].action.label,
                    onClick: error.suggestions[0].action.onClick,
                } : undefined,
            });
        } else if (error.message?.includes('timeout')) {
            // 超时错误
            const timeoutError = StoryboardError.timeout({ 
                scenesCount: scenesToExtract.length 
            });
            toast.error(timeoutError.message, {
                description: timeoutError.suggestions?.map(s => `• ${s.title}`).join('\n'),
                duration: 8000,
            });
        } else if (error.message?.includes('rate limit')) {
            // 速率限制
            const rateLimitError = StoryboardError.rateLimit();
            toast.error(rateLimitError.message, {
                description: rateLimitError.suggestions?.map(s => `• ${s.title}`).join('\n'),
                duration: 8000,
            });
        } else {
            // 未知错误
            toast.error('分镜提取失败', {
                description: error?.message || '请稍后重试',
                duration: 5000,
            });
        }
    } finally {
        onEnd();
    }
```

### 问题 1: Context 更新导致全局重渲染

**解决方案：拆分 Context**

```typescript
// src/app/pages/StoryboardEditor/context/StoryboardContext.tsx

// 拆分为 3 个独立的 Context
export const StoryboardDataContext = createContext<DataContextType | null>(null);
export const StoryboardActionsContext = createContext<ActionsContextType | null>(null);
export const StoryboardUIContext = createContext<UIContextType | null>(null);

export function StoryboardProvider({ children }: { children: React.ReactNode }) {
    const { chapterId } = useParams();
    
    // 数据层（不常变化）
    const dataValue = useMemo(() => ({
        storyboard,
        script,
        project,
        assets,
    }), [storyboard, script, project, assets]);
    
    // 操作层（稳定的函数引用）
    const actionsValue = useMemo(() => ({
        handleSave,
        handleUpdatePanel,
        handleDeletePanel,
        handleAddPanel,
        // ...
    }), [handleSave, handleUpdatePanel, handleDeletePanel, handleAddPanel]);
    
    // UI 层（频繁变化）
    const uiValue = useMemo(() => ({
        viewMode,
        selectedPanels,
        isGenerating,
        // ...
    }), [viewMode, selectedPanels, isGenerating]);
    
    return (
        <StoryboardDataContext.Provider value={dataValue}>
            <StoryboardActionsContext.Provider value={actionsValue}>
                <StoryboardUIContext.Provider value={uiValue}>
                    {children}
                </StoryboardUIContext.Provider>
            </StoryboardActionsContext.Provider>
        </StoryboardDataContext.Provider>
    );
}

// 自定义 Hooks
export function useStoryboardData() {
    const context = useContext(StoryboardDataContext);
    if (!context) throw new Error('useStoryboardData must be used within StoryboardProvider');
    return context;
}

export function useStoryboardActions() {
    const context = useContext(StoryboardActionsContext);
    if (!context) throw new Error('useStoryboardActions must be used within StoryboardProvider');
    return context;
}

export function useStoryboardUI() {
    const context = useContext(StoryboardUIContext);
    if (!context) throw new Error('useStoryboardUI must be used within StoryboardProvider');
    return context;
}
```

### 问题 2: ListView 缺少虚拟滚动

```typescript
// src/app/pages/StoryboardEditor/components/ListView.tsx

import { useVirtualizer } from '@tanstack/react-virtual';
import { useRef } from 'react';

export function ListView({ panels }: ListViewProps) {
    const parentRef = useRef<HTMLDivElement>(null);
    
    const virtualizer = useVirtualizer({
        count: panels.length,
        getScrollElement: () => parentRef.current,
        estimateSize: () => 400, // 估计每个卡片高度
        overscan: 3, // 预渲染 3 个
    });
    
    return (
        <div ref={parentRef} className="h-full overflow-auto">
            <div
                style={{
                    height: `${virtualizer.getTotalSize()}px`,
                    width: '100%',
                    position: 'relative',
                }}
            >
                {virtualizer.getVirtualItems().map((virtualItem) => {
                    const panel = panels[virtualItem.index];
                    return (
                        <div
                            key={virtualItem.key}
                            data-index={virtualItem.index}
                            ref={virtualizer.measureElement}
                            style={{
                                position: 'absolute',
                                top: 0,
                                left: 0,
                                width: '100%',
                                transform: `translateY(${virtualItem.start}px)`,
                            }}
                        >
                            <ShotCard panel={panel} index={virtualItem.index} />
                        </div>
                    );
                })}
            </div>
        </div>
    );
}
```
