```typescript
const handleAIGenerate = (type: 'fullBody' | 'face') => {
    // 1. 检查描述
    if (!character.description && !character.appearance) {
        toast.error('请先填写角色描述或外貌特征');
        return;
    }
    
    // 2. 生成提示词
    const engine = new PromptEngine(project?.directorStyle);
    let result;
    
    if (type === 'fullBody') {
        result = engine.forCharacterFullBody(character);
        onUpdate(character.id, { fullBodyPrompt: result.positive });
        toast.success('✨ 全身图提示词已生成！');
    } else {
        result = engine.forCharacterFace(character);
        onUpdate(character.id, { facePrompt: result.positive });
        toast.success('✨ 脸部图提示词已生成！');
    }
};
```

```typescript
const handleAIGenerate = () => {
    // 1. 检查描述
    if (!costume.description) {
        toast.error('请先填写服饰描述');
        return;
    }
    
    // 2. 获取关联角色
    const character = assets.characters.find(c => c.id === costume.characterId);
    
    if (!character) {
        toast.info('💡 提示：关联角色后，生成的服饰图会包含角色特征，更加准确');
    }
    
    // 3. 生成提示词
    const engine = new PromptEngine(project?.directorStyle);
    const result = engine.forCostume(costume, character);
    handleUpdateCostume(costume.id, { aiPrompt: result.positive });
    toast.success('✨ 服饰提示词已生成！');
};
```


所有生成函数都调用现有的 `PromptEngine`：

```typescript

const result = engine.forCharacterFullBody(character);
```

**优点**:
- 复用现有逻辑，无需重复开发
- 保持提示词生成的一致性
- 易于维护和升级
