### 如何集成高级搜索

```typescript
import { AdvancedSearchDialog } from '../components/AdvancedSearchDialog';

const [showAdvancedSearch, setShowAdvancedSearch] = useState(false);
const [searchFilters, setSearchFilters] = useState<AdvancedSearchFilters | null>(null);

// 应用搜索过滤
const filteredCharacters = useMemo(() => {
  let result = characters;
  
  if (searchFilters) {
    // 关键词
    if (searchFilters.keyword) {
      result = result.filter(c => 
        c.name.includes(searchFilters.keyword) ||
        c.description.includes(searchFilters.keyword)
      );
    }
    
    // 标签
    if (searchFilters.tags.length > 0) {
      result = result.filter(c => 
        c.tags?.some(tag => searchFilters.tags.includes(tag))
      );
    }
    
    // 图片状态
    if (searchFilters.hasImage !== null) {
      result = result.filter(c => 
        searchFilters.hasImage 
          ? (c.fullBodyPreview || c.facePreview)
          : (!c.fullBodyPreview && !c.facePreview)
      );
    }
    
    // 排序
    result.sort((a, b) => {
      const order = searchFilters.sortOrder === 'asc' ? 1 : -1;
      if (searchFilters.sortBy === 'name') {
        return a.name.localeCompare(b.name) * order;
      }
      // 其他排序...
    });
  }
  
  return result;
}, [characters, searchFilters]);
```

### 如何集成自动备份

```typescript
import { useAutoBackup } from '../hooks/useAutoBackup';
import { BackupManagerDialog } from '../components/BackupManagerDialog';

// 在组件中使用
const {
  createBackup,
  getBackups,
  restoreBackup,
  deleteBackup,
  exportBackup,
  importBackup,
  getStorageUsage,
} = useAutoBackup(projectId, assets);

// 显示备份管理
<BackupManagerDialog
  open={showBackupManager}
  onOpenChange={setShowBackupManager}
  backups={getBackups()}
  onRestore={(key) => {
    const data = restoreBackup(key);
    if (data) {
      setAssets(data);
    }
  }}
  onDelete={deleteBackup}
  onExport={exportBackup}
  onImport={async (file) => {
    const data = await importBackup(file);
    if (data) {
      setAssets(data);
    }
  }}
  onCreateBackup={createBackup}
  storageUsage={getStorageUsage()}
/>
```
