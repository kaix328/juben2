import type { AssetRelation } from '../types/relation';

#### 2. 测试失败

**问题描述**:
```
qualityCheck.integration.test.ts > 时长检查 > 应检测对话时长不匹配
✗ expected undefined not to be undefined
```

**影响**:
- 质量检查功能可能不完整
- 对话时长验证可能失效

**解决方案**:
```typescript
// 检查 src/app/utils/ai/qualityChecker.ts
// 确保对话时长检查返回正确的结果
export function checkDialogueDuration(shot: Shot): Issue | undefined {
  if (!shot.dialogue || shot.dialogue.length === 0) {
    return undefined;
  }
  
  // 计算对话时长
  const dialogueDuration = estimateDialogueDuration(shot.dialogue);
  const shotDuration = shot.duration || 0;
  
  if (dialogueDuration > shotDuration * 1.2) {
    return {
      type: 'warning',
      category: 'duration',
      message: '对话时长超过镜头时长',
      shotId: shot.id
    };
  }
  
  return undefined; // 确保返回值
}
```

**优先级**: 🟡 中（建议本周修复）

#### 3. 缺少环境变量配置文件

**问题描述**:

VITE_SENTRY_ENVIRONMENT=development
EOF

# 复制为本地配置
cp .env.example .env.local
# 然后编辑 .env.local 填入真实的 API 密钥
