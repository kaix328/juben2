```typescript
/**
 * 合并分镜逻辑：
 * 1. 如果是全集提取，直接使用新结果
 * 2. 如果是单集提取，保留其他集的分镜，合并新结果
 * 3. 按集数和分镜号排序
 */
const updatedPanels = episodeNumber === 'all'
  ? panelResults
  : [...otherPanels, ...panelResults].sort((a, b) => {
      // 先按集数排序
      if (a.episodeNumber !== b.episodeNumber) {
        return (a.episodeNumber || 1) - (b.episodeNumber || 1);
      }
      // 同一集内按分镜号排序
      return a.panelNumber - b.panelNumber;
    });
```
