```typescript
import { CONFIG } from '@/app/config/constants';

// 使用配置
const delay = CONFIG.PERFORMANCE.DEBOUNCE_DELAY;

```

### 2. 性能监控
```typescript
import { PerformanceMonitor } from '@/app/utils/performance';

// 监控组件渲染
useEffect(() => {
  const cleanup = PerformanceMonitor.measureRender('MyComponent');
  return cleanup;
});

// 监控函数执行
const optimizedFn = PerformanceMonitor.measureFunction('processData', processData);
```

### 3. 防抖/节流
```typescript
import { useDebounce, useThrottle } from '@/app/hooks';

// 防抖搜索
const debouncedSearchTerm = useDebounce(searchTerm, 300);

// 节流滚动
const handleScroll = useThrottle(() => {
  console.log('Scrolling...');
}, 1000);
```

### 4. 安全的定时器
```typescript
import { useInterval, useTimeout } from '@/app/hooks/useTimer';

// 自动清理的定时器
useInterval(() => {
  // 每秒执行
}, 1000);
```

### 5. 持久化缓存
```typescript
import { cacheHelpers } from '@/lib/queryClient';

// 清理缓存
cacheHelpers.clearAll();

// 获取缓存统计
const stats = cacheHelpers.getStats();
console.log(`总缓存: ${stats.total}, 活跃: ${stats.active}`);
```

## 🔧 使用指南

### 1. 使用重构后的 Hooks

```typescript
// 旧方式（仍然支持）
import { useScriptData } from '@/pages/ScriptEditor/hooks/useScriptData';

// 新方式（推荐）- 可以单独使用小 Hooks
import { 
  useScriptLoader,
  useScriptSaver,
  useScriptExtractor,
  useScriptHistory,
  useScriptEditor
} from '@/pages/ScriptEditor/hooks';

// 或者使用重构后的主 Hook（API 不变）
import { useScriptData } from '@/pages/ScriptEditor/hooks';
```

### 2. 使用配置常量

```typescript
// ❌ 不好的做法
const timer = setInterval(() => {}, 60000);

// ✅ 好的做法
import { CONFIG } from '@/app/config/constants';
const timer = setInterval(() => {}, CONFIG.BACKUP.AUTO_INTERVAL);
```

### 3. 添加性能监控

```typescript
import { PerformanceMonitor } from '@/app/utils/performance';

function MyComponent() {
  useEffect(() => {
    const cleanup = PerformanceMonitor.measureRender('MyComponent');
    return cleanup;
  });
  
  return <div>...</div>;
}
```
