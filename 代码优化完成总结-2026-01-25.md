## 🎯 使用指南

### 1. 使用配置常量

```typescript
import { CONFIG } from '@/app/config/constants';

// 使用配置
const delay = CONFIG.PERFORMANCE.DEBOUNCE_DELAY;
const maxHistory = CONFIG.BACKUP.MAX_HISTORY;
const maxChunkSize = CONFIG.AI.MAX_CHUNK_SIZE;
```

### 2. 使用重构后的Hooks

```typescript
// 方式1: 使用主Hook（推荐，API不变）
import { useScriptData } from '@/pages/ScriptEditor/hooks';

// 方式2: 单独使用小Hooks
import { 
  useScriptLoader,
  useScriptSaver,
  useScriptExtractor 
} from '@/pages/ScriptEditor/hooks';
```

### 3. 使用性能监控

```typescript
import { PerformanceMonitor } from '@/app/utils/performance';

// 监控组件渲染
useEffect(() => {
  const cleanup = PerformanceMonitor.measureRender('MyComponent');
  return cleanup;
});
```

### 4. 使用防抖/节流

```typescript
import { useDebounce, useThrottle } from '@/app/hooks';

// 防抖搜索
const debouncedSearchTerm = useDebounce(searchTerm, 300);

// 节流滚动
const handleScroll = useThrottle(() => {
  console.log('Scrolling...');
}, 1000);
```

### 5. 使用安全的定时器

```typescript
import { useInterval, useTimeout } from '@/app/hooks/useTimer';

// 自动清理的定时器
useInterval(() => {
  // 每秒执行
}, 1000);
```
