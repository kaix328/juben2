# 功能集成示例

本文档展示如何在现有的 Tab 组件中集成所有新功能。

## CharacterTab 完整集成示例

```typescript
import { useState, useMemo } from 'react';
import { Button } from '../ui/button';
import { 
  Search, 
  Plus, 
  Filter, 
  History, 
  Database, 
  BarChart3, 
  FileText,
  Network,
} from 'lucide-react';

// 导入所有新功能
import { useBatchSelection } from '../../hooks/useBatchSelection';
import { useVersionHistory } from '../../hooks/useVersionHistory';
import { useAutoBackup } from '../../hooks/useAutoBackup';
import { useTemplateSystem } from '../../hooks/useTemplateSystem';
import { useRelationGraph } from '../../hooks/useRelationGraph';
import { useAssetAnalytics } from '../../hooks/useAssetAnalytics';

import { BatchActionsBar } from '../BatchActionsBar';
import { BatchAddTagDialog } from '../BatchAddTagDialog';
import { AdvancedSearchDialog } from '../AdvancedSearchDialog';
import { VersionHistoryDialog } from '../VersionHistoryDialog';
import { BackupManagerDialog } from '../BackupManagerDialog';
import { TemplateLibraryDialog } from '../TemplateLibraryDialog';
import { RelationGraphDialog } from '../RelationGraphDialog';
import { AnalyticsDashboardDialog } from '../AnalyticsDashboardDialog';
import { VirtualGrid } from '../VirtualScroll';
import { LazyImage } from '../LazyImage';

export function CharacterTab({ projectId, characters, onUpdate, onDelete, onCreate }) {
  // ========== 状态管理 ==========
  const [searchKeyword, setSearchKeyword] = useState('');
  const [showAdvancedSearch, setShowAdvancedSearch] = useState(false);
  const [showBatchTagDialog, setShowBatchTagDialog] = useState(false);
  const [showVersionHistory, setShowVersionHistory] = useState(false);
  const [showBackupManager, setShowBackupManager] = useState(false);
  const [showTemplateLibrary, setShowTemplateLibrary] = useState(false);
  const [showRelationGraph, setShowRelationGraph] = useState(false);
  const [showAnalytics, setShowAnalytics] = useState(false);
  const [selectedCharacterId, setSelectedCharacterId] = useState<string | null>(null);
  const [searchFilters, setSearchFilters] = useState(null);

  // ========== Hooks ==========
  // 批量选择
  const {
    selectedIds,
    isBatchMode,
    toggleBatchMode,
    toggleSelect,
    selectAll,
    clearSelection,
    isSelected,
    selectedCount,
  } = useBatchSelection(characters.length);

  // 版本历史
  const { saveVersion, getVersions, rollbackToVersion } = useVersionHistory(projectId);

  // 自动备份
  const {
    createBackup,
    getBackups,
    restoreBackup,
    deleteBackup,
    exportBackup,
    importBackup,
    getStorageUsage,
  } = useAutoBackup(projectId, { characters, scenes: [], props: [], costumes: [] });

  // 模板系统
  const {
    getAllTemplates,
    getCategories,
    applyTemplate,
    createTemplateFromAsset,
    deleteTemplate,
    exportTemplate,
    importTemplate,
  } = useTemplateSystem(projectId);

  // 关系图谱
  const {
    exportGraphData,
  } = useRelationGraph(projectId);

  // 数据分析
  const usageMap = new Map(); // 实际使用时从分镜数据中计算
  const analytics = useAssetAnalytics(
    { characters, scenes: [], props: [], costumes: [] },
    usageMap
  );

  // ========== 搜索和过滤 ==========
  const filteredCharacters = useMemo(() => {
    let result = characters;

    // 应用高级搜索过滤
    if (searchFilters) {
      if (searchFilters.keyword) {
        result = result.filter(c =>
          c.name.includes(searchFilters.keyword) ||
          c.description?.includes(searchFilters.keyword)
        );
      }
      if (searchFilters.tags.length > 0) {
        result = result.filter(c =>
          c.tags?.some(tag => searchFilters.tags.includes(tag))
        );
      }
      if (searchFilters.hasImage !== null) {
        result = result.filter(c =>
          searchFilters.hasImage
            ? (c.fullBodyPreview || c.facePreview)
            : (!c.fullBodyPreview && !c.facePreview)
        );
      }
    }

    // 简单搜索
    if (searchKeyword && !searchFilters) {
      result = result.filter(c =>
        c.name.toLowerCase().includes(searchKeyword.toLowerCase())
      );
    }

    return result;
  }, [characters, searchKeyword, searchFilters]);

  // ========== 事件处理 ==========
  // 更新角色（带版本历史）
  const handleUpdate = (id: string, updates: Partial<Character>) => {
    const character = characters.find(c => c.id === id);
    if (character) {
      // 保存版本
      saveVersion(id, 'character', character, updates);
      // 执行更新
      onUpdate(id, updates);
    }
  };

  // 批量删除
  const handleBatchDelete = () => {
    if (confirm(`确定要删除选中的 ${selectedCount} 个角色吗？`)) {
      selectedIds.forEach(id => onDelete(id));
      clearSelection();
    }
  };

  // 批量添加标签
  const handleBatchAddTag = (tags: string[]) => {
    selectedIds.forEach(id => {
      const character = characters.find(c => c.id === id);
      if (character) {
        const newTags = [...new Set([...(character.tags || []), ...tags])];
        handleUpdate(id, { tags: newTags });
      }
    });
    clearSelection();
  };

  // 应用模板
  const handleApplyTemplate = (templateId: string) => {
    const templateData = applyTemplate(templateId);
    if (templateData) {
      onCreate(templateData);
    }
  };

  // 从当前角色创建模板
  const handleCreateTemplate = () => {
    if (!selectedCharacterId) return;
    const character = characters.find(c => c.id === selectedCharacterId);
    if (character) {
      const name = prompt('模板名称:');
      const description = prompt('模板描述:');
      const category = prompt('模板分类:');
      if (name && description && category) {
        createTemplateFromAsset(character, name, description, category, 'character');
      }
    }
  };

  // 获取所有标签
  const allTags = useMemo(() => {
    const tags = new Set<string>();
    characters.forEach(c => c.tags?.forEach(tag => tags.add(tag)));
    return Array.from(tags);
  }, [characters]);

  // ========== 渲染 ==========
  return (
    <div className="space-y-4">
      {/* 工具栏 */}
      <div className="flex items-center gap-2">
        <div className="relative flex-1">
          <Search className="absolute left-3 top-1/2 -translate-y-1/2 w-4 h-4 text-gray-400" />
          <input
            type="text"
            placeholder="搜索角色..."
            value={searchKeyword}
            onChange={(e) => setSearchKeyword(e.target.value)}
            className="w-full pl-9 pr-4 py-2 border rounded-lg"
          />
        </div>
        
        <Button variant="outline" onClick={() => setShowAdvancedSearch(true)}>
          <Filter className="w-4 h-4 mr-2" />
          高级搜索
        </Button>

        <Button variant="outline" onClick={toggleBatchMode}>
          批量操作
        </Button>

        <Button variant="outline" onClick={() => setShowTemplateLibrary(true)}>
          <FileText className="w-4 h-4 mr-2" />
          模板
        </Button>

        <Button onClick={() => onCreate({})}>
          <Plus className="w-4 h-4 mr-2" />
          新建角色
        </Button>
      </div>

      {/* 次级工具栏 */}
      <div className="flex items-center gap-2">
        <Button variant="ghost" size="sm" onClick={() => setShowRelationGraph(true)}>
          <Network className="w-4 h-4 mr-2" />
          关系图谱
        </Button>

        <Button variant="ghost" size="sm" onClick={() => setShowAnalytics(true)}>
          <BarChart3 className="w-4 h-4 mr-2" />
          数据分析
        </Button>

        <Button variant="ghost" size="sm" onClick={() => setShowBackupManager(true)}>
          <Database className="w-4 h-4 mr-2" />
          备份管理
        </Button>

        {selectedCharacterId && (
          <Button
            variant="ghost"
            size="sm"
            onClick={() => setShowVersionHistory(true)}
          >
            <History className="w-4 h-4 mr-2" />
            版本历史
          </Button>
        )}
      </div>

      {/* 角色列表 - 使用虚拟滚动 */}
      <VirtualGrid
        items={filteredCharacters}
        itemWidth={280}
        itemHeight={360}
        containerWidth={1200}
        containerHeight={600}
        gap={16}
        renderItem={(character, index) => (
          <div
            className={`
              border rounded-lg p-4 cursor-pointer transition-all
              ${isSelected(character.id) ? 'border-blue-500 bg-blue-50' : 'hover:shadow-lg'}
            `}
            onClick={() => {
              if (isBatchMode) {
                toggleSelect(character.id);
              } else {
                setSelectedCharacterId(character.id);
              }
            }}
          >
            {/* 批量选择复选框 */}
            {isBatchMode && (
              <input
                type="checkbox"
                checked={isSelected(character.id)}
                onChange={() => toggleSelect(character.id)}
                className="absolute top-2 left-2"
              />
            )}

            {/* 角色图片 - 使用懒加载 */}
            {character.fullBodyPreview && (
              <LazyImage
                src={character.fullBodyPreview}
                alt={character.name}
                className="w-full h-48 object-cover rounded"
              />
            )}

            {/* 角色信息 */}
            <h3 className="font-semibold mt-2">{character.name}</h3>
            <p className="text-sm text-gray-500 line-clamp-2">
              {character.description}
            </p>

            {/* 标签 */}
            {character.tags && character.tags.length > 0 && (
              <div className="flex flex-wrap gap-1 mt-2">
                {character.tags.slice(0, 3).map(tag => (
                  <span key={tag} className="text-xs bg-gray-100 px-2 py-1 rounded">
                    {tag}
                  </span>
                ))}
              </div>
            )}
          </div>
        )}
      />

      {/* 批量操作栏 */}
      <BatchActionsBar
        selectedCount={selectedCount}
        totalCount={characters.length}
        isAllSelected={selectedIds.size === characters.length}
        onSelectAll={() => selectAll(characters.map(c => c.id))}
        onClearSelection={clearSelection}
        onBatchDelete={handleBatchDelete}
        onBatchAddTag={() => setShowBatchTagDialog(true)}
      />

      {/* 对话框 */}
      <AdvancedSearchDialog
        open={showAdvancedSearch}
        onOpenChange={setShowAdvancedSearch}
        onSearch={setSearchFilters}
        availableTags={allTags}
        currentFilters={searchFilters}
      />

      <BatchAddTagDialog
        open={showBatchTagDialog}
        onOpenChange={setShowBatchTagDialog}
        onConfirm={handleBatchAddTag}
        selectedCount={selectedCount}
        existingTags={allTags}
      />

      <VersionHistoryDialog
        open={showVersionHistory}
        onOpenChange={setShowVersionHistory}
        versions={selectedCharacterId ? getVersions(selectedCharacterId) : []}
        onRollback={(versionId) => {
          if (selectedCharacterId) {
            const data = rollbackToVersion(versionId, selectedCharacterId);
            if (data) {
              handleUpdate(selectedCharacterId, data);
            }
          }
        }}
        assetName={
          selectedCharacterId
            ? characters.find(c => c.id === selectedCharacterId)?.name || ''
            : ''
        }
      />

      <BackupManagerDialog
        open={showBackupManager}
        onOpenChange={setShowBackupManager}
        backups={getBackups()}
        onRestore={(key) => {
          const data = restoreBackup(key);
          if (data) {
            // 恢复数据
          }
        }}
        onDelete={deleteBackup}
        onExport={exportBackup}
        onImport={importBackup}
        onCreateBackup={createBackup}
        storageUsage={getStorageUsage()}
      />

      <TemplateLibraryDialog
        open={showTemplateLibrary}
        onOpenChange={setShowTemplateLibrary}
        templates={getAllTemplates('character')}
        categories={getCategories('character')}
        onApply={handleApplyTemplate}
        onDelete={deleteTemplate}
        onExport={exportTemplate}
        onImport={importTemplate}
        onCreateFromCurrent={handleCreateTemplate}
        type="character"
      />

      <RelationGraphDialog
        open={showRelationGraph}
        onOpenChange={setShowRelationGraph}
        {...exportGraphData(characters, [], [], [])}
        onNodeClick={(nodeId, nodeType) => {
          if (nodeType === 'character') {
            setSelectedCharacterId(nodeId);
          }
        }}
      />

      <AnalyticsDashboardDialog
        open={showAnalytics}
        onOpenChange={setShowAnalytics}
        analytics={analytics}
        onExportReport={() => {
          // 导出报表逻辑
        }}
      />
    </div>
  );
}
```

## 关键集成点

### 1. 批量操作
- 使用 `useBatchSelection` Hook 管理选择状态
- 在卡片上显示复选框（批量模式下）
- 使用 `BatchActionsBar` 显示浮动工具栏

### 2. 版本历史
- 在每次更新时调用 `saveVersion`
- 通过 `VersionHistoryDialog` 查看和回滚

### 3. 自动备份
- Hook 自动处理备份（每5分钟 + 页面卸载前）
- 通过 `BackupManagerDialog` 管理备份

### 4. 高级搜索
- 使用 `AdvancedSearchDialog` 设置过滤条件
- 在 `useMemo` 中应用过滤逻辑

### 5. 模板系统
- 从模板创建新资产
- 从现有资产创建模板
- 导入/导出模板

### 6. 性能优化
- 使用 `VirtualGrid` 处理大量数据
- 使用 `LazyImage` 延迟加载图片
- 使用 `useMemo` 缓存计算结果

### 7. 数据分析
- 使用 `useAssetAnalytics` 计算统计数据
- 通过 `AnalyticsDashboardDialog` 展示分析结果

### 8. 关系图谱
- 使用 `useRelationGraph` 管理关系数据
- 通过 `RelationGraphDialog` 可视化展示

## 性能优化建议

1. **虚拟滚动**: 当资产数量 > 100 时启用
2. **图片懒加载**: 所有图片都应使用 LazyImage
3. **useMemo**: 缓存过滤和排序结果
4. **防抖**: 搜索输入使用防抖
5. **分页**: 考虑添加分页功能

## 数据验证

在创建和更新时使用验证：

```typescript
import { validateCharacter, ValidationError } from '../../utils/validation';

const handleCreate = (data: Partial<Character>) => {
  try {
    validateCharacter(data);
    onCreate(data);
  } catch (error) {
    if (error instanceof ValidationError) {
      alert(`验证失败: ${error.message}`);
    }
  }
};
```
